# Helper functions

def first_order_geoImageParameters(focalLength, objDistance, objHeight=None):
    """Returns first order approximations of image distance, image height, and lateral magnification
    using Gaussian lens equation.
    """
    imgHeight = None
    imgDistance = focalLength*objDistance/(objDistance - focalLength)
    latMag = imgDistance/objDistance
    if objHeight:
        imgHeight = latMag*objHeight
    return imgDistance, imgHeight, latMag

def incoherent_diffraction_limited_cutoff(wxp, zxp, lamda):
    """
    Returns the diffraction limited incoherent cutoff frequency in cycles/mm
    
    Parameters
    ----------
    wxp : float; exit pupil radius
    zxp : float; exit pupil distance
    lamda : float; wavelength
    
    Return
    ------
    diff_ltd_cutoff_freq
    
    """
    return (2.0*wxp)/(lamda*zxp)

def diff_ltd_sim_criterion_satisfied(wxp, zxp, lamda, L, num_samples):
    """verifies whether or not diffraction-limited imaging simulation is satisfied
    
    Parameters
    ----------
    wxp : float; exit pupil radius
    zxp : float; exit pupil distance
    lamda : float; wavelength
    L : float; image plane side (assigning physical units to the geometric image)
    num_samples : integer, number of samples along one dimension of the image, assuming square image 
    
    See: Computational Fourier Optics by David Voelz (equ. 7.17, & sec. 8.71 for details)
    """
    two_f0 = incoherent_diffraction_limited_cutoff(wxp, zxp, lamda)
    du = L/num_samples
    nyquist_freq = 1/(2.0*du)
    return two_f0 < nyquist_freq

To change the amount of defocus, change obj_offset below...use the multiplier in front of the number .... 0 for no defocus, -1 and 1.

# ----------------------------------
# aberration coefficient
# waves of defocus defocus (w020 is in terms of wavelength) ... this needs to be multiplied by
# the wavelength

# create the frequency grid Fu, Fv, for the fourier transform
fu = tu.fft_freq_bin(M, order=1, Fs=1/du) # same as np.arange(-1/(2*du),1/(2*du), 1/L) used in the book.
fv = fu  # since it is a square grid with equal number of samples along each axis
Fu, Fv = np.meshgrid(fu, fv)
wxp = aperture_r
k = 2*np.pi/lamda

# wavefront (for the axial image point 0,0) 
W = fou.seidel_5(0, 0, -lamda*zxp*Fu/wxp, -lamda*zxp*Fv/wxp, wd=w020*lamda)

# coherent transfer function
H = siu.circR(np.hypot(Fu, Fv)/f0)*np.exp(-1j*k*W)

# the autocorrelation is the inverseFFT of the power-spectrum, which is the square
# of the fft of the function.
OTF = np.fft.ifft2(np.abs(np.fft.fft2(np.fft.fftshift(H))**2)) 
OTF = OTF/OTF.max() # since the maximum occurs at dc

# Incoherent imaging PSF
