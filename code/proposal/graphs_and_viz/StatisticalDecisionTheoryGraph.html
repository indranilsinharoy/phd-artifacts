from __future__ import division
from scipy.stats import binom
from scipy import interpolate

# Figure (7) -- Statistical decision theory concept (using normal distributions) 
N_normal = 1000
n_normal = linspace(0,1,N_normal)
# Using the definition of Gaussian distribution
# as the norm function plots the standard normal
def my_gaussian(mu, sigma, x):
    dist = 1.0/(sigma*np.sqrt(2.0*np.pi))*np.exp(-(x-mu)**2 / (2.0*sigma**2))
    return dist

Y_normal1 = my_gaussian(0.3, 0.08, n_normal)
Y_normal2 = my_gaussian(0.5, 0.08, n_normal)


# Figure (11) -- Binomial distributions
N1 = 480   # Number of trials
p1 = 0.45  # 

N2 = 40
p2 = 0.084 

X1 = binom(N1, p1)
X2 = binom(N2, p2)

x1 = linspace(0,N1,N1+1)
x2 = linspace(0,N2,N2+1)

Y1 = X1.pmf(x1)
Y2 = X2.pmf(x2)

# Combine the two distributions
f = interpolate.interp1d(x2, Y2)
x2_new = linspace(0, N2, N1+1)
Y2_new = f(x2_new)

# Need to appropiately scale Y1 as the number of trials are different
